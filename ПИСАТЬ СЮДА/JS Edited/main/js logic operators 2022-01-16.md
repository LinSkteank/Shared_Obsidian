# –õ–æ–≥–∏—á–µ—Å–∫–∏–µ –æ–ø–µ—Ä–∞—Ç–æ—Ä—ã

–í JavaScript –µ—Å—Ç—å —Ç—Ä–∏ –ª–æ–≥–∏—á–µ—Å–∫–∏—Ö –æ–ø–µ—Ä–∞—Ç–æ—Ä–∞: `||` (–ò–õ–ò), `&&` (–ò) –∏ `!` (–ù–ï).

## [|| (–ò–õ–ò)](https://learn.javascript.ru/logical-operators#ili)

–û–ø–µ—Ä–∞—Ç–æ—Ä ¬´–ò–õ–ò¬ª –≤—ã–≥–ª—è–¥–∏—Ç –∫–∞–∫ –¥–≤–æ–π–Ω–æ–π —Å–∏–º–≤–æ–ª –≤–µ—Ä—Ç–∏–∫–∞–ª—å–Ω–æ–π —á–µ—Ä—Ç—ã:

`result = a || b;`

–¢—Ä–∞–¥–∏—Ü–∏–æ–Ω–Ω–æ –≤ –ø—Ä–æ–≥—Ä–∞–º–º–∏—Ä–æ–≤–∞–Ω–∏–∏ –ò–õ–ò –ø—Ä–µ–¥–Ω–∞–∑–Ω–∞—á–µ–Ω–æ —Ç–æ–ª—å–∫–æ –¥–ª—è –º–∞–Ω–∏–ø—É–ª–∏—Ä–æ–≤–∞–Ω–∏—è –±—É–ª–µ–≤—ã–º–∏ –∑–Ω–∞—á–µ–Ω–∏—è–º–∏: –≤ —Å–ª—É—á–∞–µ, –µ—Å–ª–∏ –∫–∞–∫–æ–π-–ª–∏–±–æ –∏–∑ –∞—Ä–≥—É–º–µ–Ω—Ç–æ–≤ `true`, –æ–Ω –≤–µ—Ä–Ω—ë—Ç `true`, –≤ –ø—Ä–æ—Ç–∏–≤–æ–ø–æ–ª–æ–∂–Ω–æ–π —Å–∏—Ç—É–∞—Ü–∏–∏ –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç—Å—è `false`.

–°—É—â–µ—Å—Ç–≤—É–µ—Ç –≤—Å–µ–≥–æ —á–µ—Ç—ã—Ä–µ –≤–æ–∑–º–æ–∂–Ω—ã–µ –ª–æ–≥–∏—á–µ—Å–∫–∏–µ –∫–æ–º–±–∏–Ω–∞—Ü–∏–∏:
`alert( true || true );   // true
alert( false || true );  // true
alert( true || false );  // true
alert( false || false ); // false`

–ö–∞–∫ –º—ã –º–æ–∂–µ–º –Ω–∞–±–ª—é–¥–∞—Ç—å, —Ä–µ–∑—É–ª—å—Ç–∞—Ç –æ–ø–µ—Ä–∞—Ü–∏–π –≤—Å–µ–≥–¥–∞ —Ä–∞–≤–µ–Ω `true`, –∑–∞ –∏—Å–∫–ª—é—á–µ–Ω–∏–µ–º —Å–ª—É—á–∞—è, –∫–æ–≥–¥–∞ –æ–±–∞ –∞—Ä–≥—É–º–µ–Ω—Ç–∞ `false`.

–ï—Å–ª–∏ –∑–Ω–∞—á–µ–Ω–∏–µ –Ω–µ –ª–æ–≥–∏—á–µ—Å–∫–æ–≥–æ —Ç–∏–ø–∞, —Ç–æ –æ–Ω–æ –∫ –Ω–µ–º—É –ø—Ä–∏–≤–æ–¥–∏—Ç—Å—è –≤ —Ü–µ–ª—è—Ö –≤—ã—á–∏—Å–ª–µ–Ω–∏–π.

–ù–∞–ø—Ä–∏–º–µ—Ä, —á–∏—Å–ª–æ `1` –±—É–¥–µ—Ç –≤–æ—Å–ø—Ä–∏–Ω—è—Ç–æ –∫–∞–∫ `true`, –∞ `0` ‚Äì –∫–∞–∫ `false`:
```javascript
`if (1 || 0) { // —Ä–∞–±–æ—Ç–∞–µ—Ç –∫–∞–∫ if( true || false )
  alert( 'truthy!' );
}`
```
–û–±—ã—á–Ω–æ –æ–ø–µ—Ä–∞—Ç–æ—Ä `||` –∏—Å–ø–æ–ª—å–∑—É–µ—Ç—Å—è –≤ `if` –¥–ª—è –ø—Ä–æ–≤–µ—Ä–∫–∏ –∏—Å—Ç–∏–Ω–Ω–æ—Å—Ç–∏ –ª—é–±–æ–≥–æ –∏–∑ –∑–∞–¥–∞–Ω–Ω—ã—Ö —É—Å–ª–æ–≤–∏–π.

```javascript
`let hour = 9;

if (hour < 10 || hour > 18) {
  alert( '–û—Ñ–∏—Å –∑–∞–∫—Ä—ã—Ç.' );
}`

`let hour = 12;
let isWeekend = true;

if (hour < 10 || hour > 18 || isWeekend) {
  alert( '–û—Ñ–∏—Å –∑–∞–∫—Ä—ã—Ç.' ); // —ç—Ç–æ –≤—ã—Ö–æ–¥–Ω–æ–π
}`
```
–ü—Ä–∏ –≤—ã–ø–æ–ª–Ω–µ–Ω–∏–∏ –ò–õ–ò || —Å –Ω–µ—Å–∫–æ–ª—å–∫–∏–º–∏ –∑–Ω–∞—á–µ–Ω–∏—è–º–∏:

`result = value1 || value2 || value3;`

–û–ø–µ—Ä–∞—Ç–æ—Ä `||` –≤—ã–ø–æ–ª–Ω—è–µ—Ç —Å–ª–µ–¥—É—é—â–∏–µ –¥–µ–π—Å—Ç–≤–∏—è:

-   –í—ã—á–∏—Å–ª—è–µ—Ç –æ–ø–µ—Ä–∞–Ω–¥—ã —Å–ª–µ–≤–∞ –Ω–∞–ø—Ä–∞–≤–æ.
-   –ö–∞–∂–¥—ã–π –æ–ø–µ—Ä–∞–Ω–¥ –∫–æ–Ω–≤–µ—Ä—Ç–∏—Ä—É–µ—Ç –≤ –ª–æ–≥–∏—á–µ—Å–∫–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ. –ï—Å–ª–∏ —Ä–µ–∑—É–ª—å—Ç–∞—Ç `true`, –æ—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞–µ—Ç—Å—è –∏ –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç –∏—Å—Ö–æ–¥–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ —ç—Ç–æ–≥–æ –æ–ø–µ—Ä–∞–Ω–¥–∞.
-   –ï—Å–ª–∏ –≤—Å–µ –æ–ø–µ—Ä–∞–Ω–¥—ã —è–≤–ª—è—é—Ç—Å—è –ª–æ–∂–Ω—ã–º–∏ (`false`), –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç –ø–æ—Å–ª–µ–¥–Ω–∏–π –∏–∑ –Ω–∏—Ö.

–ó–Ω–∞—á–µ–Ω–∏–µ –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç—Å—è –≤ –∏—Å—Ö–æ–¥–Ω–æ–º –≤–∏–¥–µ, –±–µ–∑ –ø—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏—è.

–î—Ä—É–≥–∏–º–∏ —Å–ª–æ–≤–∞–º–∏, —Ü–µ–ø–æ—á–∫–∞ –ò–õ–ò `"||"` –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç –ø–µ—Ä–≤–æ–µ –∏—Å—Ç–∏–Ω–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ –∏–ª–∏ –ø–æ—Å–ª–µ–¥–Ω–µ–µ, –µ—Å–ª–∏ —Ç–∞–∫–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ –Ω–µ –Ω–∞–π–¥–µ–Ω–æ.
```javascript
`alert( 1 || 0 ); // 1
alert( true || 'no matter what' ); // true

alert( null || 1 ); // 1 (–ø–µ—Ä–≤–æ–µ –∏—Å—Ç–∏–Ω–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ)
alert( null || 0 || 1 ); // 1 (–ø–µ—Ä–≤–æ–µ –∏—Å—Ç–∏–Ω–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ)
alert( undefined || null || 0 ); // 0 (–ø–æ—Å–∫–æ–ª—å–∫—É –≤—Å–µ –ª–æ–∂–Ω–æ, –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç—Å—è –ø–æ—Å–ª–µ–¥–Ω–µ–µ –∑–Ω–∞—á–µ–Ω–∏–µ)`
```
–≠—Ç–æ –¥–µ–ª–∞–µ—Ç –≤–æ–∑–º–æ–∂–Ω—ã–º –±–æ–ª–µ–µ –∏–Ω—Ç–µ—Ä–µ—Å–Ω–æ–µ –ø—Ä–∏–º–µ–Ω–µ–Ω–∏–µ –æ–ø–µ—Ä–∞—Ç–æ—Ä–∞ –ø–æ —Å—Ä–∞–≤–Ω–µ–Ω–∏—é —Å ¬´—á–∏—Å—Ç—ã–º, —Ç—Ä–∞–¥–∏—Ü–∏–æ–Ω–Ω—ã–º, —Ç–æ–ª—å–∫–æ –±—É–ª–µ–≤—ã–º –ò–õ–ò¬ª.

1.  **–ü–æ–ª—É—á–µ–Ω–∏–µ –ø–µ—Ä–≤–æ–≥–æ –∏—Å—Ç–∏–Ω–Ω–æ–≥–æ –∑–Ω–∞—á–µ–Ω–∏—è –∏–∑ —Å–ø–∏—Å–∫–∞ –ø–µ—Ä–µ–º–µ–Ω–Ω—ã—Ö –∏–ª–∏ –≤—ã—Ä–∞–∂–µ–Ω–∏–π.**
    
    –ü—Ä–µ–¥—Å—Ç–∞–≤–∏–º, —á—Ç–æ —É –Ω–∞—Å –∏–º–µ–µ—Ç—Å—è —Ä—è–¥ –ø–µ—Ä–µ–º–µ–Ω–Ω—ã—Ö, –∫–æ—Ç–æ—Ä—ã–µ –º–æ–≥—É—Ç —Å–æ–¥–µ—Ä–∂–∞—Ç—å –¥–∞–Ω–Ω—ã–µ –∏–ª–∏ –±—ã—Ç—å `null/undefined`. –ö–∞–∫ –º—ã –º–æ–∂–µ–º –Ω–∞–π—Ç–∏ –ø–µ—Ä–≤—É—é –ø–µ—Ä–µ–º–µ–Ω–Ω—É—é —Å –¥–∞–Ω–Ω—ã–º–∏?
    ```javascript
    `let currentUser = null;
    let defaultUser = "John";
    
    let name = currentUser || defaultUser || "unnamed";
    
    alert( name ); // –≤—ã–±–∏—Ä–∞–µ—Ç—Å—è "John" ‚Äì –ø–µ—Ä–≤–æ–µ –∏—Å—Ç–∏–Ω–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ`
    
    –ï—Å–ª–∏ –±—ã –∏ `currentUser`, –∏ `defaultUser` –±—ã–ª–∏ –ª–æ–∂–Ω—ã–º–∏, –≤ –∫–∞—á–µ—Å—Ç–≤–µ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–∞ –º—ã –±—ã –Ω–∞–±–ª—é–¥–∞–ª–∏ `"unnamed"`.
    ```
2.  **–°–æ–∫—Ä–∞—â—ë–Ω–Ω–æ–µ –≤—ã—á–∏—Å–ª–µ–Ω–∏–µ.**
    
    –û–ø–µ—Ä–∞–Ω–¥–∞–º–∏ –º–æ–≥—É—Ç –±—ã—Ç—å –∫–∞–∫ –æ—Ç–¥–µ–ª—å–Ω—ã–µ –∑–Ω–∞—á–µ–Ω–∏—è, —Ç–∞–∫ –∏ –ø—Ä–æ–∏–∑–≤–æ–ª—å–Ω—ã–µ –≤—ã—Ä–∞–∂–µ–Ω–∏—è. –ò–õ–ò –≤—ã—á–∏—Å–ª—è–µ—Ç –∏—Ö —Å–ª–µ–≤–∞ –Ω–∞–ø—Ä–∞–≤–æ. –í—ã—á–∏—Å–ª–µ–Ω–∏–µ –æ—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞–µ—Ç—Å—è –ø—Ä–∏ –¥–æ—Å—Ç–∏–∂–µ–Ω–∏–∏ –ø–µ—Ä–≤–æ–≥–æ –∏—Å—Ç–∏–Ω–Ω–æ–≥–æ –∑–Ω–∞—á–µ–Ω–∏—è. –≠—Ç–æ—Ç –ø—Ä–æ—Ü–µ—Å—Å –Ω–∞–∑—ã–≤–∞–µ—Ç—Å—è ¬´—Å–æ–∫—Ä–∞—â—ë–Ω–Ω—ã–º –≤—ã—á–∏—Å–ª–µ–Ω–∏–µ–º¬ª, –ø–æ—Å–∫–æ–ª—å–∫—É –≤—Ç–æ—Ä–æ–π –æ–ø–µ—Ä–∞–Ω–¥ –≤—ã—á–∏—Å–ª—è–µ—Ç—Å—è —Ç–æ–ª—å–∫–æ –≤ —Ç–æ–º —Å–ª—É—á–∞–µ, –µ—Å–ª–∏ –ø–µ—Ä–≤–æ–≥–æ –Ω–µ–¥–æ—Å—Ç–∞—Ç–æ—á–Ω–æ –¥–ª—è –≤—ã—á–∏—Å–ª–µ–Ω–∏—è –≤—Å–µ–≥–æ –≤—ã—Ä–∞–∂–µ–Ω–∏—è.
    
    –≠—Ç–æ —Ö–æ—Ä–æ—à–æ –∑–∞–º–µ—Ç–Ω–æ, –∫–æ–≥–¥–∞ –≤—ã—Ä–∞–∂–µ–Ω–∏–µ, —É–∫–∞–∑–∞–Ω–Ω–æ–µ –≤ –∫–∞—á–µ—Å—Ç–≤–µ –≤—Ç–æ—Ä–æ–≥–æ –∞—Ä–≥—É–º–µ–Ω—Ç–∞, –∏–º–µ–µ—Ç –ø–æ–±–æ—á–Ω—ã–π —ç—Ñ—Ñ–µ–∫—Ç, –Ω–∞–ø—Ä–∏–º–µ—Ä, –∏–∑–º–µ–Ω–µ–Ω–∏–µ –ø–µ—Ä–µ–º–µ–Ω–Ω–æ–π.
    
    –í –ø—Ä–∏–≤–µ–¥—ë–Ω–Ω–æ–º –Ω–∏–∂–µ –ø—Ä–∏–º–µ—Ä–µ `x` –Ω–µ –∏–∑–º–µ–Ω—è–µ—Ç—Å—è:
    ```javascript
    `let x;
    
    true || (x = 1);
    
    alert(x); // undefined, –ø–æ—Ç–æ–º—É —á—Ç–æ (x = 1) –Ω–µ –≤—ã—á–∏—Å–ª—è–µ—Ç—Å—è`
    ```
    –ï—Å–ª–∏ –±—ã –ø–µ—Ä–≤—ã–π –∞—Ä–≥—É–º–µ–Ω—Ç –∏–º–µ–ª –∑–Ω–∞—á–µ–Ω–∏–µ `false`, —Ç–æ `||` –ø—Ä–∏—Å—Ç—É–ø–∏–ª –±—ã –∫ –≤—ã—á–∏—Å–ª–µ–Ω–∏—é –≤—Ç–æ—Ä–æ–≥–æ –∏ –≤—ã–ø–æ–ª–Ω–∏–ª –æ–ø–µ—Ä–∞—Ü–∏—é –ø—Ä–∏—Å–≤–∞–∏–≤–∞–Ω–∏—è:
    `let x;
    
    false || (x = 1);
    
    alert(x); // 1`

## [&& (–ò)](https://learn.javascript.ru/logical-operators#i)

–û–ø–µ—Ä–∞—Ç–æ—Ä –ò –ø–∏—à–µ—Ç—Å—è –∫–∞–∫ –¥–≤–∞ –∞–º–ø–µ—Ä—Å–∞–Ω–¥–∞ `&&`:

`result = a && b;`

–í —Ç—Ä–∞–¥–∏—Ü–∏–æ–Ω–Ω–æ–º –ø—Ä–æ–≥—Ä–∞–º–º–∏—Ä–æ–≤–∞–Ω–∏–∏ –ò –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç `true`, –µ—Å–ª–∏ –æ–±–∞ –∞—Ä–≥—É–º–µ–Ω—Ç–∞ –∏—Å—Ç–∏–Ω–Ω—ã, –∞ –∏–Ω–∞—á–µ ‚Äì `false`:

`alert( true && true );   // true
alert( false && true );  // false
alert( true && false );  // false
alert( false && false ); // false`

–ü—Ä–∏–º–µ—Ä —Å `if`:
```javascript
`let hour = 12;
let minute = 30;

if (hour == 12 && minute == 30) {
  alert( 'The time is 12:30' );
}`
```
## [–ò ¬´&&¬ª –Ω–∞—Ö–æ–¥–∏—Ç –ø–µ—Ä–≤–æ–µ –ª–æ–∂–Ω–æ–µ¬†–∑–Ω–∞—á–µ–Ω–∏–µ](https://learn.javascript.ru/logical-operators#i-nahodit-pervoe-lozhnoeznachenie)

–ü—Ä–∏ –Ω–µ—Å–∫–æ–ª—å–∫–∏—Ö –ø–æ–¥—Ä—è–¥ –æ–ø–µ—Ä–∞—Ç–æ—Ä–∞—Ö –ò:

`result = value1 && value2 && value3;`

–û–ø–µ—Ä–∞—Ç–æ—Ä `&&` –≤—ã–ø–æ–ª–Ω—è–µ—Ç —Å–ª–µ–¥—É—é—â–∏–µ –¥–µ–π—Å—Ç–≤–∏—è:

-   –í—ã—á–∏—Å–ª—è–µ—Ç –æ–ø–µ—Ä–∞–Ω–¥—ã —Å–ª–µ–≤–∞ –Ω–∞–ø—Ä–∞–≤–æ.
-   –ö–∞–∂–¥—ã–π –æ–ø–µ—Ä–∞–Ω–¥ –ø—Ä–µ–æ–±—Ä–∞–∑—É–µ—Ç –≤ –ª–æ–≥–∏—á–µ—Å–∫–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ. –ï—Å–ª–∏ —Ä–µ–∑—É–ª—å—Ç–∞—Ç `false`, –æ—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞–µ—Ç—Å—è –∏ –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç –∏—Å—Ö–æ–¥–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ —ç—Ç–æ–≥–æ –æ–ø–µ—Ä–∞–Ω–¥–∞.
-   –ï—Å–ª–∏ –≤—Å–µ –æ–ø–µ—Ä–∞–Ω–¥—ã –±—ã–ª–∏ –∏—Å—Ç–∏–Ω–Ω—ã–º–∏, –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç—Å—è –ø–æ—Å–ª–µ–¥–Ω–∏–π.

–î—Ä—É–≥–∏–º–∏ —Å–ª–æ–≤–∞–º–∏, –ò –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç –ø–µ—Ä–≤–æ–µ –ª–æ–∂–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ. –ò–ª–∏ –ø–æ—Å–ª–µ–¥–Ω–µ–µ, –µ—Å–ª–∏ –Ω–∏—á–µ–≥–æ –Ω–µ –Ω–∞–π–¥–µ–Ω–æ.

–í—ã—à–µ—É–∫–∞–∑–∞–Ω–Ω—ã–µ –ø—Ä–∞–≤–∏–ª–∞ —Å—Ö–æ–∂–∏ —Å –ø–æ–≤–µ–¥–µ–Ω–∏–µ–º –ò–õ–ò. –†–∞–∑–Ω–∏—Ü–∞ –≤ —Ç–æ–º, —á—Ç–æ –ò –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç –ø–µ—Ä–≤–æ–µ _–ª–æ–∂–Ω–æ–µ_ –∑–Ω–∞—á–µ–Ω–∏–µ, –∞ –ò–õ–ò‚Ää‚Äì‚Ää –ø–µ—Ä–≤–æ–µ _–∏—Å—Ç–∏–Ω–Ω–æ–µ_.

–ú–æ–∂–Ω–æ –ø–µ—Ä–µ–¥–∞—Ç—å –Ω–µ—Å–∫–æ–ª—å–∫–æ –∑–Ω–∞—á–µ–Ω–∏–π –ø–æ–¥—Ä—è–¥. –í —Ç–∞–∫–æ–º —Å–ª—É—á–∞–µ –≤–æ–∑–≤—Ä–∞—Ç–∏—Ç—Å—è –ø–µ—Ä–≤–æ–µ ¬´–ª–æ–∂–Ω–æ–µ¬ª –∑–Ω–∞—á–µ–Ω–∏–µ, –Ω–∞ –∫–æ—Ç–æ—Ä–æ–º –æ—Å—Ç–∞–Ω–æ–≤–∏–ª–∏—Å—å –≤—ã—á–∏—Å–ª–µ–Ω–∏—è.


`alert( 1 && 2 && null && 3 ); // null`

**–ö–æ–≥–¥–∞ –≤—Å–µ –∑–Ω–∞—á–µ–Ω–∏—è –≤–µ—Ä–Ω—ã, –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç—Å—è –ø–æ—Å–ª–µ–¥–Ω–µ–µ**

`alert( 1 && 2 && 3 ); // 3`

–ü—Ä–∏–æ—Ä–∏—Ç–µ—Ç –æ–ø–µ—Ä–∞—Ç–æ—Ä–∞ `&&` –±–æ–ª—å—à–µ, —á–µ–º —É `||`

## [! (–ù–ï)](https://learn.javascript.ru/logical-operators#ne)

–û–ø–µ—Ä–∞—Ç–æ—Ä –ù–ï –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ–Ω –≤–æ—Å–∫–ª–∏—Ü–∞—Ç–µ–ª—å–Ω—ã–º –∑–Ω–∞–∫–æ–º `!`.

–°–∏–Ω—Ç–∞–∫—Å–∏—Å –¥–æ–≤–æ–ª—å–Ω–æ –ø—Ä–æ—Å—Ç:

`result = !value;`

–û–ø–µ—Ä–∞—Ç–æ—Ä –ø—Ä–∏–Ω–∏–º–∞–µ—Ç –æ–¥–∏–Ω –∞—Ä–≥—É–º–µ–Ω—Ç –∏ –≤—ã–ø–æ–ª–Ω—è–µ—Ç —Å–ª–µ–¥—É—é—â–∏–µ –¥–µ–π—Å—Ç–≤–∏—è:

1.  –°–Ω–∞—á–∞–ª–∞ –ø—Ä–∏–≤–æ–¥–∏—Ç –∞—Ä–≥—É–º–µ–Ω—Ç –∫ –ª–æ–≥–∏—á–µ—Å–∫–æ–º—É —Ç–∏–ø—É `true/false`.
2.  –ó–∞—Ç–µ–º –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç –ø—Ä–æ—Ç–∏–≤–æ–ø–æ–ª–æ–∂–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ.

`alert( !true ); // false
alert( !0 ); // true`

–í —á–∞—Å—Ç–Ω–æ—Å—Ç–∏, –¥–≤–æ–π–Ω–æ–µ –ù–ï –∏—Å–ø–æ–ª—å–∑—É—é—Ç –¥–ª—è –ø—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏—è –∑–Ω–∞—á–µ–Ω–∏–π –∫ –ª–æ–≥–∏—á–µ—Å–∫–æ–º—É —Ç–∏–ø—É:

`alert( !!"non-empty string" ); // true
alert( !!null ); // false`

–¢–æ –µ—Å—Ç—å –ø–µ—Ä–≤–æ–µ –ù–ï –ø—Ä–µ–æ–±—Ä–∞–∑—É–µ—Ç –∑–Ω–∞—á–µ–Ω–∏–µ –≤ –ª–æ–≥–∏—á–µ—Å–∫–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ –∏ –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç –æ–±—Ä–∞—Ç–Ω–æ–µ, –∞ –≤—Ç–æ—Ä–æ–µ –ù–ï —Å–Ω–æ–≤–∞ –∏–Ω–≤–µ—Ä—Ç–∏—Ä—É–µ—Ç –µ–≥–æ. –í –∫–æ–Ω—Ü–µ –º—ã –∏–º–µ–µ–º –ø—Ä–æ—Å—Ç–æ–µ –ø—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏–µ –∑–Ω–∞—á–µ–Ω–∏—è –≤ –ª–æ–≥–∏—á–µ—Å–∫–æ–µ.

–ï—Å—Ç—å –Ω–µ–º–Ω–æ–≥–æ –±–æ–ª–µ–µ –ø–æ–¥—Ä–æ–±–Ω—ã–π —Å–ø–æ—Å–æ–± —Å–¥–µ–ª–∞—Ç—å —Ç–æ –∂–µ —Å–∞–º–æ–µ ‚Äì –≤—Å—Ç—Ä–æ–µ–Ω–Ω–∞—è —Ñ—É–Ω–∫—Ü–∏—è `Boolean`:

`alert( Boolean("non-empty string") ); // true
alert( Boolean(null) ); // false`

–ü—Ä–∏–æ—Ä–∏—Ç–µ—Ç –ù–ï `!` —è–≤–ª—è–µ—Ç—Å—è –Ω–∞–∏–≤—ã—Å—à–∏–º –∏–∑ –≤—Å–µ—Ö –ª–æ–≥–∏—á–µ—Å–∫–∏—Ö –æ–ø–µ—Ä–∞—Ç–æ—Ä–æ–≤, –ø–æ—ç—Ç–æ–º—É –æ–Ω –≤—Å–µ–≥–¥–∞ –≤—ã–ø–æ–ª–Ω—è–µ—Ç—Å—è –ø–µ—Ä–≤—ã–º, –ø–µ—Ä–µ–¥ `&&` –∏–ª–∏ `||`.





# –û–ø–µ—Ä–∞—Ç–æ—Ä –æ–±—ä–µ–¥–∏–Ω–µ–Ω–∏—è —Å null '??'

–û–ø–µ—Ä–∞—Ç–æ—Ä –æ–±—ä–µ–¥–∏–Ω–µ–Ω–∏—è —Å null –ø—Ä–µ–¥—Å—Ç–∞–≤–ª—è–µ—Ç —Å–æ–±–æ–π –¥–≤–∞ –≤–æ–ø—Ä–æ—Å–∏—Ç–µ–ª—å–Ω—ã—Ö –∑–Ω–∞–∫–∞ `??`.

–†–µ–∑—É–ª—å—Ç–∞—Ç –≤—ã—Ä–∞–∂–µ–Ω–∏—è `a ?? b` –±—É–¥–µ—Ç —Å–ª–µ–¥—É—é—â–∏–º:

-   `a`, –µ—Å–ª–∏ –∑–Ω–∞—á–µ–Ω–∏–µ `a` –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–æ,
-   `b`, –µ—Å–ª–∏ –∑–Ω–∞—á–µ–Ω–∏–µ `a` –Ω–µ –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–æ.

–¢–æ –µ—Å—Ç—å –æ–ø–µ—Ä–∞—Ç–æ—Ä `??` –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç –ø–µ—Ä–≤—ã–π –∞—Ä–≥—É–º–µ–Ω—Ç, –µ—Å–ª–∏ –æ–Ω –Ω–µ `null/undefined`, –∏–Ω–∞—á–µ –≤—Ç–æ—Ä–æ–π.

–û–ø–µ—Ä–∞—Ç–æ—Ä –æ–±—ä–µ–¥–∏–Ω–µ–Ω–∏—è —Å null –Ω–µ —è–≤–ª—è–µ—Ç—Å—è —á–µ–º-—Ç–æ –ø—Ä–∏–Ω—Ü–∏–ø–∏–∞–ª—å–Ω–æ –Ω–æ–≤—ã–º. –≠—Ç–æ –≤—Å–µ–≥–æ –ª–∏—à—å —É–¥–æ–±–Ω—ã–π —Å–∏–Ω—Ç–∞–∫—Å–∏—Å, –∫–∞–∫ –∏–∑ –¥–≤—É—Ö –∑–Ω–∞—á–µ–Ω–∏–π –ø–æ–ª—É—á–∏—Ç—å –æ–¥–Ω–æ ¬´–æ–ø—Ä–µ–¥–µ–ª—ë–Ω–Ω–æ–µ¬ª.

–í–æ—Ç –∫–∞–∫ –º–æ–∂–Ω–æ –ø–µ—Ä–µ–ø–∏—Å–∞—Ç—å –≤—ã—Ä–∞–∂–µ–Ω–∏–µ `result = a ?? b`, –∏—Å–ø–æ–ª—å–∑—É—è —É–∂–µ –∑–Ω–∞–∫–æ–º—ã–µ –Ω–∞–º –æ–ø–µ—Ä–∞—Ç–æ—Ä—ã:

`result = (a !== null && a !== undefined) ? a : b;`

–ö–∞–∫ –ø—Ä–∞–≤–∏–ª–æ, –æ–ø–µ—Ä–∞—Ç–æ—Ä `??` –Ω—É–∂–µ–Ω –¥–ª—è —Ç–æ–≥–æ, —á—Ç–æ–±—ã –∑–∞–¥–∞—Ç—å –∑–Ω–∞—á–µ–Ω–∏–µ –ø–æ —É–º–æ–ª—á–∞–Ω–∏—é –¥–ª—è –ø–æ—Ç–µ–Ω—Ü–∏–∞–ª—å–Ω–æ –Ω–µ–æ–ø—Ä–µ–¥–µ–ª—ë–Ω–Ω–æ–π –ø–µ—Ä–µ–º–µ–Ω–Ω–æ–π.

`let user;

alert(user ?? "–ê–Ω–æ–Ω–∏–º"); // –ê–Ω–æ–Ω–∏–º`

–ö–æ–Ω–µ—á–Ω–æ, –µ—Å–ª–∏ –±—ã –ø–µ—Ä–µ–º–µ–Ω–Ω–∞—è `user` —Å–æ–¥–µ—Ä–∂–∞–ª–∞ –ª—é–±–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ, –∫—Ä–æ–º–µ `null/undefined`, —Ç–æ –º—ã –±—ã —É–≤–∏–¥–µ–ª–∏ –µ–≥–æ:

`let user = "–ò–≤–∞–Ω";

alert(user ?? "–ê–Ω–æ–Ω–∏–º"); // –ò–≤–∞–Ω`

## [–°—Ä–∞–≤–Ω–µ–Ω–∏–µ —Å ||](https://learn.javascript.ru/nullish-coalescing-operator#sravnenie-s)

–û–ø–µ—Ä–∞—Ç–æ—Ä –ò–õ–ò `||` –º–æ–∂–Ω–æ –∏—Å–ø–æ–ª—å–∑–æ–≤–∞—Ç—å –¥–ª—è —Ç–æ–≥–æ –∂–µ, —á—Ç–æ –∏ `??`

–ù–∞–ø—Ä–∏–º–µ—Ä, –µ—Å–ª–∏ –≤ –ø—Ä–∏–≤–µ–¥—ë–Ω–Ω–æ–º –≤—ã—à–µ –∫–æ–¥–µ –∑–∞–º–µ–Ω–∏—Ç—å `??` –Ω–∞ `||`, —Ç–æ –±—É–¥–µ—Ç —Ç–æ—Ç –∂–µ —Å–∞–º—ã–π —Ä–µ–∑—É–ª—å—Ç–∞—Ç:
```javascript
`let firstName = null;
let lastName = null;
let nickName = "–°—É–ø–µ—Ä–∫–æ–¥–µ—Ä";

// –ø–æ–∫–∞–∑—ã–≤–∞–µ—Ç –ø–µ—Ä–≤–æ–µ –∏—Å—Ç–∏–Ω–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ:
_alert(firstName || lastName || nickName || "–ê–Ω–æ–Ω–∏–º"); // –°—É–ø–µ—Ä–∫–æ–¥–µ—Ä_`
```
–û–ø–µ—Ä–∞—Ç–æ—Ä –ò–õ–ò `||` —Å—É—â–µ—Å—Ç–≤—É–µ—Ç —Å —Å–∞–º–æ–≥–æ –ø–æ—è–≤–ª–µ–Ω–∏—è JavaScript, –ø–æ—ç—Ç–æ–º—É —Ä–∞–Ω–µ–µ –¥–ª—è —Ä–µ—à–µ–Ω–∏—è –ø–æ—Ö–æ–∂–∏—Ö –∑–∞–¥–∞—á —Ä–∞–∑—Ä–∞–±–æ—Ç—á–∏–∫–∏ –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–ª–∏ –∏–º–µ–Ω–Ω–æ –µ–≥–æ.

–° –¥—Ä—É–≥–æ–π —Å—Ç–æ—Ä–æ–Ω—ã, —Å—Ä–∞–≤–Ω–∏—Ç–µ–ª—å–Ω–æ –Ω–µ–¥–∞–≤–Ω–æ –≤ —è–∑—ã–∫ –±—ã–ª –¥–æ–±–∞–≤–ª–µ–Ω –æ–ø–µ—Ä–∞—Ç–æ—Ä –æ–±—ä–µ–¥–∏–Ω–µ–Ω–∏—è —Å null `??` –∫–∞–∫ —Ä–∞–∑ –ø–æ—Ç–æ–º—É, —á—Ç–æ –º–Ω–æ–≥–∏–µ –±—ã–ª–∏ –Ω–µ–¥–æ–≤–æ–ª—å–Ω—ã –æ–ø–µ—Ä–∞—Ç–æ—Ä–æ–º `||`.

–í–∞–∂–Ω–æ–µ —Ä–∞–∑–ª–∏—á–∏–µ –º–µ–∂–¥—É –Ω–∏–º–∏ –∑–∞–∫–ª—é—á–∞–µ—Ç—Å—è –≤ —Ç–æ–º, —á—Ç–æ:

-   `||` –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç –ø–µ—Ä–≤–æ–µ _–∏—Å—Ç–∏–Ω–Ω–æ–µ_ –∑–Ω–∞—á–µ–Ω–∏–µ.
-   `??` –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç –ø–µ—Ä–≤–æ–µ _–æ–ø—Ä–µ–¥–µ–ª—ë–Ω–Ω–æ–µ_ –∑–Ω–∞—á–µ–Ω–∏–µ.

–ü—Ä–æ—â–µ –≥–æ–≤–æ—Ä—è, –æ–ø–µ—Ä–∞—Ç–æ—Ä `||` –Ω–µ —Ä–∞–∑–ª–∏—á–∞–µ—Ç `false`, `0`, –ø—É—Å—Ç—É—é —Å—Ç—Ä–æ–∫—É `""` –∏ `null/undefined`. –î–ª—è –Ω–µ–≥–æ –æ–Ω–∏ –≤—Å–µ –æ–¥–∏–Ω–∞–∫–æ–≤—ã–µ, —Ç.–µ. —è–≤–ª—è—é—Ç—Å—è –ª–æ–∂–Ω—ã–º–∏ –∑–Ω–∞—á–µ–Ω–∏—è–º–∏. –ï—Å–ª–∏ –ø–µ—Ä–≤—ã–º –∞—Ä–≥—É–º–µ–Ω—Ç–æ–º –¥–ª—è –æ–ø–µ—Ä–∞—Ç–æ—Ä–∞ `||` –±—É–¥–µ—Ç –ª—é–±–æ–µ –∏–∑ –ø–µ—Ä–µ—á–∏—Å–ª–µ–Ω–Ω—ã—Ö –∑–Ω–∞—á–µ–Ω–∏–π, —Ç–æ –≤ –∫–∞—á–µ—Å—Ç–≤–µ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–∞ –º—ã –ø–æ–ª—É—á–∏–º –≤—Ç–æ—Ä–æ–π –∞—Ä–≥—É–º–µ–Ω—Ç.

–û–¥–Ω–∞–∫–æ –Ω–∞ –ø—Ä–∞–∫—Ç–∏–∫–µ —á–∞—Å—Ç–æ —Ç—Ä–µ–±—É–µ—Ç—Å—è –∏—Å–ø–æ–ª—å–∑–æ–≤–∞—Ç—å –∑–Ω–∞—á–µ–Ω–∏–µ –ø–æ —É–º–æ–ª—á–∞–Ω–∏—é —Ç–æ–ª—å–∫–æ —Ç–æ–≥–¥–∞, –∫–æ–≥–¥–∞ –ø–µ—Ä–µ–º–µ–Ω–Ω–∞—è —è–≤–ª—è–µ—Ç—Å—è `null/undefined`. –í–µ–¥—å –∏–º–µ–Ω–Ω–æ —Ç–æ–≥–¥–∞ –∑–Ω–∞—á–µ–Ω–∏–µ –¥–µ–π—Å—Ç–≤–∏—Ç–µ–ª—å–Ω–æ –Ω–µ–∏–∑–≤–µ—Å—Ç–Ω–æ/–Ω–µ –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–æ.

`let height = 0;

alert(height ?? 100); // 0`

-   `height || 100` –ø—Ä–æ–≤–µ—Ä—è–µ—Ç, –∏–º–µ–µ—Ç –ª–∏ –ø–µ—Ä–µ–º–µ–Ω–Ω–∞—è `height` –ª–æ–∂–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ, —á—Ç–æ —Ç–∞–∫ –∏ –µ—Å—Ç—å,
    -   –ø–æ—ç—Ç–æ–º—É —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–º —è–≤–ª—è–µ—Ç—Å—è –≤—Ç–æ—Ä–æ–π –∞—Ä–≥—É–º–µ–Ω—Ç, —Ç.–µ. `100`.
-   `height ?? 100` –ø—Ä–æ–≤–µ—Ä—è–µ—Ç, —á—Ç–æ –ø–µ—Ä–µ–º–µ–Ω–Ω–∞—è `height` —Å–æ–¥–µ—Ä–∂–∏—Ç `null/undefined`, –∞ –ø–æ—Å–∫–æ–ª—å–∫—É —ç—Ç–æ –Ω–µ —Ç–∞–∫,
    -   —Ç–æ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–º —è–≤–ª—è–µ—Ç—Å—è —Å–∞–º–∞ –ø–µ—Ä–µ–º–µ–Ω–Ω–∞—è `height`, —Ç.–µ. `0`.

–ï—Å–ª–∏ –Ω—É–ª–µ–≤–∞—è –≤—ã—Å–æ—Ç–∞ —è–≤–ª—è–µ—Ç—Å—è ¬´–Ω–æ—Ä–º–∞–ª—å–Ω—ã–º¬ª –∑–Ω–∞—á–µ–Ω–∏–µ–º, –∫–æ—Ç–æ—Ä–æ–µ –Ω–µ –¥–æ–ª–∂–Ω–æ –∑–∞–º–µ–Ω—è—Ç—å—Å—è –∑–Ω–∞—á–µ–Ω–∏–µ–º –ø–æ —É–º–æ–ª—á–∞–Ω–∏—é, —Ç–æ –æ–ø–µ—Ä–∞—Ç–æ—Ä `??` –¥–µ–ª–∞–µ—Ç –∫–∞–∫ —Ä–∞–∑ —Ç–æ, —á—Ç–æ –Ω—É–∂–Ω–æ.
